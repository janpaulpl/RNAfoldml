let to_dot_string r =
  if r |> Secondary.get_pairs |> Secondary.is_pknot then
    Invalid_argument "Cannot convert pseudoknot to dot" |> raise
  else
    r |> Secondary.get_pairs
    |> Array.mapi (fun i j ->
           if j = ~-1 then "." else if i < j then "(" else ")")
    |> Array.fold_left ( ^ ) ""
